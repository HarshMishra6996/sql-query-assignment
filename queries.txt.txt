-- Create Users Table
CREATE TABLE users (
    user_id INT PRIMARY KEY,
    name VARCHAR(50)
);

-- Create Orders Table
CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    user_id INT,
    amount INT
);

-- Insert sample data
INSERT INTO users VALUES
(1,'Alice'),
(2,'Bob'),
(3,'Charlie');

INSERT INTO orders VALUES
(101,1,500),
(102,1,200),
(103,2,300);

-- INNER JOIN example
SELECT users.name, orders.amount
FROM users
INNER JOIN orders
ON users.user_id = orders.user_id;

-- LEFT JOIN example
SELECT users.name, orders.amount
FROM users
LEFT JOIN orders
ON users.user_id = orders.user_id;
